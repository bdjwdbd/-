```typescript
// INT4 量化：每个值用 4 bit 表示（-8 到 7）
function quantizeToInt4(value: number): number {
  return Math.max(-8, Math.min(7, Math.round(value * 8)));
}

// INT4 点积（使用 Int8Array 存储，每字节存 2 个 INT4）
function int4DotProduct(a: Uint8Array, b: Uint8Array): number {
  let sum = 0;
  for (let i = 0; i < a.length; i++) {
    const aHigh = (a[i] >> 4) - 8;  // 高 4 bit
    const aLow = (a[i] & 0x0F) - 8; // 低 4 bit
    const bHigh = (b[i] >> 4) - 8;
    const bLow = (b[i] & 0x0F) - 8;
    sum += aHigh * bHigh + aLow * bLow;
  }
  return sum;
}
```

## 八、风险与应对

| 风险 | 应对 |
|------|------|
| INT4 精度损失 | 提供可配置的 INT8/INT4 切换 |
| HNSW 召回率不足 | 提供可配置的 efSearch 参数 |
| 内存不足 | 提供磁盘持久化选项 |
| 线程数过多 | 自动限制最大线程数 |
